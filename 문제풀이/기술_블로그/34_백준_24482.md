import sys

sys.setrecursionlimit(10 ** 6)
input = sys.stdin.readline

def dfs(start, count):
    visited[start] = count
        
    for cur in graph[start]:
        if visited[cur] == -1:
            dfs(cur, count + 1)

    graph[a].append(b)
    graph[b].append(a)
