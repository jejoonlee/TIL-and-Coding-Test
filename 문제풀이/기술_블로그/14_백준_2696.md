import heapq
import math

T = int(input())

for _ in range(T):

    N = int(input())

    nums = []
    for _ in range(int(math.ceil(N / 10))):
        nums += list(map(int, input().split()))

    left, right = [], []
    middle = nums[0]
    
    result = [nums[0]]      

    for i in range(1, N):

        else:

                middle = -heapq.heappop(left)

            elif len(left) < len(right):
                heapq.heappush(left, -middle)
                middle = heapq.heappop(right)

            result.append(middle)

    print(len(result))
