        answer = 0
    else:
        for num in range(2, n + 1):
            cnt = 0
            
            for i in range(1, (num // 2) + 1):

                    cnt += 1
                    
                    if cnt > 1:
                        flag = False
                        break
                        
            if flag == True:
                answer += 1
            else:
                flag = True
    
    return answer
