N = int(input())
work = []
for _ in range(N):
    work.append(tuple(map(int, input().split())))

work.sort(key=lambda x: x[1])
hour = 0
flag = True

while True:
    time = hour
    
    for i in range(len(work)):
        time += work[i][0]

        if time > work[i][1]:
            flag = False
            break
    
    if flag == True:
        hour += 1
    else:
        break

if hour != 0:
    print(hour - 1)
else:
    print(-1)
