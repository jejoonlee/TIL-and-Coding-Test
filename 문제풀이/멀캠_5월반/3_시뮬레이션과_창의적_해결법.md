```python
N, K = map(int, input().split())

dr = [-1, 0, 0, 1]
dc = [0, -1, 1, 0]
field = [[0] * N for _ in range(N)]

for _ in range(K):
	y, x = map(int, input().split())
	y, x = y - 1, x - 1
	field[y][x] += 1
	
	for i in range(4):
		sr = dr[i] + y
		sc = dc[i] + x
		
		if 0 <= sr < N and 0 <= sc < N:
			field[sr][sc] += 1
			
answer = 0

for f in field:
	answer += sum(f)

print(answer)
```
