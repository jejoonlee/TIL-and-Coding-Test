```python
from collections import deque

money, M = map(int, input().split())

reserve = deque([])

for _ in range(M):
	command, value = input().split()
	value = int(value)
	
	if command == "deposit":
		money += value
	
	elif command == "pay":
		if value <= money:
			money -= value
			
	elif command == "reservation":
		if len(reserve) == 0 and value <= money:
			money -= value
		else:
			reserve.append(value)
			
	while reserve:
		if money >= reserve[0]:
			money -= reserve.popleft()
		else:
			break
			
print(money)
```
